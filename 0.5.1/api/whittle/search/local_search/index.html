    Bases: FIFOScheduler

    Local Search Scheduler for hyperparameter optimization.

    This scheduler uses a local search strategy to explore the configuration space. It extends the FIFOScheduler and integrates with the LocalSearch searcher.

    PARAMETER DESCRIPTION
    config_space

    Configuration space for the evaluation function.

    TYPE: dict[str, Any]

    metric

    List of metrics to optimize.

    TYPE: list[str]

    mode

    TYPE: list[str] | str DEFAULT: 'min'

    start_point

    Starting point for the search. Defaults to None.

    TYPE: dict[str, Any] | None DEFAULT: None

    random_seed

    Random seed for reproducibility. Defaults to None.

    TYPE: int | None DEFAULT: None

    points_to_evaluate

    Initial points to evaluate. Defaults to None.

    TYPE: list[dict] | None DEFAULT: None

    Additional arguments for the FIFOScheduler.

    def __init__(
    self,
    config_space: dict[str, Any],
    metric: list[str] | str,
    points_to_evaluate: list[dict] | None = None,
    start_point: dict | None = None,
    mode: list[str] | str = "min",
    **kwargs: Any,
):
    if start_point is None:
        self.start_point = {
            k: v.sample() if isinstance(v, Domain) else v
            for k, v in config_space.items()
        }
    else:
        self.start_point = start_point

    self._pareto_front: list[PopulationElement] = []

    if points_to_evaluate is None:
        points_to_evaluate = [self.start_point]
    else:
        points_to_evaluate.append(self.start_point)

    super().__init__(
        config_space,
        metric,
        mode=mode,
        points_to_evaluate=points_to_evaluate,
        **kwargs,
    )
    if isinstance(self._mode, list):
        self._metric_op: dict[str, Any] = {
            metric: 1 if mode == "min" else -1
            for metric, mode in zip(metric, self._mode)
        }
    else:
        if self._mode == "min":
            self._metric_op = dict(zip(self._metric, [1.0] * len(self._metric)))
        elif self._mode == "max":
            self._metric_op = dict(zip(self._metric, [-1.0] * len(self._metric)))
